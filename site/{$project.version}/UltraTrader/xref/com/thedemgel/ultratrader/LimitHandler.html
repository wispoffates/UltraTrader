204 	private static ConfigValue getConfigValue(ConfigurationSection section, String search) {
205 		if (section.contains(search)) {
206 			return new ConfigValue(section.get(search));
207 		}
208 
209 		// Attempt to get the default config.
210 		// If not set, throw error and return null.
211 		section = getDefaultsSection();
212 
213 		if (section.contains(search)) {
214 			return new ConfigValue(section.get(search));
215 		}
216 
217 		if(UltraTrader.getInstance().isDebug()) {
218 			Bukkit.getLogger().log(Level.WARNING, "Config value not found in limits.yml in default section: " + search);
219 		}
220 		return null;
221 	}
