292 		private StoreConfig config;
293 		private Shop shop;

295 		public SaveShop(Shop shop) {
296 			config = new StoreConfig(UltraTrader.getInstance(), new File(STORE_DIR, shop.getId() + ".yml"));
297 			this.shop = shop;
298 		}
299 
300 		@Override
301 		public void run() {
302             ConfigurationSerialization.registerClass(CategoryItem.class);
303             ConfigurationSection catSection = config.getConfig().createSection("category");
304 
305             for (CategoryItem categoryItem : shop.getCategoryItem().values()) {
306                 catSection.set(categoryItem.getCategoryId() + ".id", categoryItem.getCategoryId());
307                 catSection.set(categoryItem.getCategoryId() + ".itemStack", categoryItem);
308                 catSection.set(categoryItem.getCategoryId() + ".slot", categoryItem.getSlot());
309                 catSection.set(categoryItem.getCategoryId() + ".lore", categoryItem.getLore());
310             }
311 
312             ConfigurationSection priceConfig = config.getConfig().createSection("priceList");
313 
314             for (ItemPrice priceList : shop.getPriceList().values()) {
315                 priceConfig.set(priceList.getId() + ".description", priceList.getDescription());
316                 priceConfig.set(priceList.getId() + ".category", priceList.getCategoryId());
317                 priceConfig.set(priceList.getId() + ".itemStack", priceList.getItemStack());
318                 priceConfig.set(priceList.getId() + ".sellPrice", priceList.getSellPrice());
319                 priceConfig.set(priceList.getId() + ".buyPrice", priceList.getBuyPrice());
320                 priceConfig.set(priceList.getId() + ".random", priceList.getId());
321                 priceConfig.set(priceList.getId() + ".slot", priceList.getSlot());
322             }
323 
324 			ConfigurationSection invConfig = config.getConfig().getConfigurationSection("inventory");
325 			if (invConfig == null) {
326 				invConfig = config.getConfig().createSection("inventory");
327 			}
328 
329 			// Save Inventory INFO
330 			ConfigurationSection inventoryInfoConfig = invConfig.getConfigurationSection("info");
331 			if (inventoryInfoConfig == null) {
332 				inventoryInfoConfig = invConfig.createSection("info");
333 				ConfigValue<String> defaultinventory = new ConfigValue(InventoryInterfaceHandler.DEFAULT_INVENTORY_TYPE);
334 				inventoryInfoConfig.set("type", defaultinventory.getValue());
335 			}
336 
337 			for (Entry<String, ConfigValue> inventoryinfo : shop.getInventoryinfo().entrySet()) {
338 				inventoryInfoConfig.set(inventoryinfo.getKey(), inventoryinfo.getValue().getValue());
339 			}
340 
341 			// Save Inventory items
342 			ConfigurationSection invconf = invConfig.createSection("items");
343 
344 			Integer count = 0;
345 
346 			for (Map.Entry<ItemStack, Integer> entry : shop.getInventory().entrySet()) {
347 				invconf.set(count.toString() + ".itemstack", entry.getKey());
348 				invconf.set(count.toString() + ".amount", entry.getValue());
349 				count++;
350 			}
351 
352 			// Get wallet information
353 			ConfigurationSection walletconfig = config.getConfig().getConfigurationSection("wallet");
354 			if (walletconfig == null) {
355 				walletconfig = config.getConfig().createSection("wallet");
356 				walletconfig.set("type", WalletHandler.DEFAULT_WALLET_TYPE);
357 			}
358 
359 			for (Entry<String, ConfigValue> walletinfo : shop.getWalletinfo().entrySet()) {
360 				walletconfig.set(walletinfo.getKey(), walletinfo.getValue().getValue());
361 			}
362 
363 			ConfigurationSection infoconfig = config.getConfig().getConfigurationSection("info");
364 			if (infoconfig == null) {
365 				infoconfig = config.getConfig().createSection("info");
366 				//ConfigValue<String> defaultname = new ConfigValue(L.getString("general.newshopname"));
367 				infoconfig.set("name", new ConfigValue(L.getString("general.newshopname")));
368 			}
369 
370 			for (Entry<String, ConfigValue> info : shop.getInfo().entrySet()) {
371 				infoconfig.set(info.getKey(), info.getValue().getValue());
372 			}
373 
374 			ConfigurationSection blockConfig = config.getConfig().getConfigurationSection("blocks");
375             if (blockConfig == null) {
376                 blockConfig = config.getConfig().createSection("blocks");
377             }
378 
379             ConfigurationSection blocks = blockConfig.createSection("blocklocation");
380 
381             int i = 0;
382             for (Location loc : shop.getBlockShops()) {
383                 blocks.set(i + ".world", loc.getWorld().getName());
384                 blocks.set(i + ".x", loc.getBlockX());
385                 blocks.set(i + ".y", loc.getBlockY());
386                 blocks.set(i + ".z", loc.getBlockZ());
387                 i++;
388             }
389 
390 			config.saveConfig();
391 		}
