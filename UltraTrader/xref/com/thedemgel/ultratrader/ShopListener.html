291 		if (event.getView() instanceof ShopInventoryView) {
292 			final ShopInventoryView view = (ShopInventoryView) event.getView();
293 
294             final HumanEntity player = event.getPlayer();
295 
296             Bukkit.getScheduler().runTaskLater(plugin, new Runnable() {
297                 @Override
298                 public void run() {
299                     ConcurrentMap<String, ItemPrice> items = view.getShop().getPriceList();
300                     //ConcurrentMap<String, ItemPrice> sellitems = view.getShop().getSellPrices();
301 
302                     for (ItemStack item : player.getInventory()) {
303                         String id = view.getShop().getItemId(item);
304                         boolean remove = false;
305                         if (id != "000000") {
306                             if (items.containsKey(id)) {
307                                 remove = true;
308                             }
309                             if (remove) {
310                                 player.getInventory().remove(item);
311                             }
312                         }
313                     }
314 
315                     if (!view.isKeepAlive()) {
316                         UltraTrader.getStoreHandler().getInventoryHandler().removeInventoryView((Player) event.getPlayer());
317                     }
318                 }
319             }, UltraTrader.BUKKIT_SCHEDULER_DELAY);
320 		}
321 
322 		if (event.getView() instanceof AdminItemPlacementView || event.getView() instanceof AdminCategoryPlacementView) {
323 			final Player player = (Player) event.getPlayer();
324 
325 			Bukkit.getScheduler().scheduleSyncDelayedTask(plugin, new Runnable() {
326 
327 				@Override
328 				public void run() {
329 					InventoryHandler handler = UltraTrader.getStoreHandler().getInventoryHandler();
330 
331 					if (handler.hasInventoryView(player)) {
332 						ShopInventoryView view = (ShopInventoryView) handler.getInventoryView(player);
333 						view.setKeepAlive(false);

374 							convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_NPC, npc);
375                             convo.getContext().setSessionData(ConversationHandler.CONVERSATION_SESSION_IS_BLOCK, false);

377 						} else {
378 							player.sendRawMessage(ChatColor.RED + L.getString("conversation.error.inconvo"));
379 						}

383 						return;

421 		}

424

430 			Block block = event.getClickedBlock();

433                 // TODO: open shop

435

438                         return;
439                     }
440                 }
441 
442                 InventoryHandler handler = UltraTrader.getStoreHandler().getInventoryHandler();
443                 // Open Store
444                 if (handler.hasInventoryView(player)) {
445                     ShopInventoryView view = (ShopInventoryView) handler.getInventoryView(player);
446                     if (!view.getShop().getId().equals(shopId)) {
447                         handler.createBuyInventoryView(player, UltraTrader.getStoreHandler().getShop(shopId));
448                         // TODO: change to BLOCK
449                         //view.setTarget(npc);

451                 } else {
452                     handler.createBuyInventoryView(player, UltraTrader.getStoreHandler().getShop(shopId));
453                     // TODO: change to BLOCK
454                     //.setTarget(npc);
455                 }
456 
457                 handler.openInventory(player);
458 			}
459 
460             // Check if item in hand is create block shop item
461             if (item != null && item.getType().equals(Material.PAPER)) {
462                 if (BlockShopHandler.createBlockShop(player, event.getClickedBlock())) {
463                     return;
464                 }
465             }
466 		}
467 
468 		if (event.getAction().equals(Action.RIGHT_CLICK_AIR) || event.getAction().equals(Action.RIGHT_CLICK_BLOCK)) {
469 			item = event.getItem();
470 
471 			// after here is all itemsinhand -- if no item in hand. return.

473 				return;
474 			}

478 			 }*/
479 			if (StoreItem.isUltraTraderItem(item)) {

481 				String id = StoreItem.getItemShopId(item);
482 				UltraTrader.getStoreHandler().getInventoryHandler().createBuyInventoryView(player, UltraTrader.getStoreHandler().getShop(Integer.valueOf(id)));
483 				UltraTrader.getStoreHandler().getInventoryHandler().openInventory(player);
