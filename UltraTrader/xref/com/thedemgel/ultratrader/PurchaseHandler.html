20  /**
21   * If there is anything left (no order is required) a reverse order will be
22   * done, and price will be reduced by anything that remains.
23   *
24   * Order for secure transactions:
25   *
26   * 1) figure price for all items in "Buy Box" - "Buy Stack" 2) Check if player
27   * has available funds 3) Withdraw funds - hold funds in temp variable * 4)
28   * Remove items (also check for availability) from traders inventory * 5) If
29   * difference, remove from players "Buy Box" 6) Attempt to put all items in
30   * players inventory 7) Any remaining items not put into inventory - figure
31   * price value 8) subtract remaining value from holding 9) refund funds to
32   * player account 10) deposit funds into trader account 11) return any remaining
33   * items to trader inventory.
34   */

64  	public static void processPurchase(Shop shop, Player player, ItemStack buyStack) {
65  
66  		boolean charge = true;
67  		// Should we charge
68  		if (shop.isOwner(player)) {
69  			charge = false;
70  		}
71  
72  		String buyStackId = shop.getItemId(buyStack);
73  		ItemPrice invItem = shop.getPriceList().get(buyStackId);
74  		BigDecimal buyStackPriceEach = invItem.getSellPrice();
75  		BigDecimal buyStackPrice = buyStackPriceEach.multiply(BigDecimal.valueOf(buyStack.getAmount()));
76          ItemStack baseItem = invItem.getItemStack().clone();
77          baseItem.setAmount(buyStack.getAmount());
78  
79          PlayerPurchaseFromShopEvent event = new PlayerPurchaseFromShopEvent(player, invItem, shop, baseItem, buyStackPrice);
80          Bukkit.getServer().getPluginManager().callEvent(event);
81  
82          if (event.isCancelled()) {
83              return;
84          }
85  
86  		EconomyResponse heldFunds;
87  		if (!charge) {
88  			heldFunds = new EconomyResponse(0, 0, ResponseType.SUCCESS, "");
89  		} else if (UltraTrader.getEconomy().has(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue())) {
90  			heldFunds = UltraTrader.getEconomy().withdrawPlayer(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue());
91  		} else {
92  			player.sendMessage(L.getString("transaction.sale.player.notenoughfunds"));
93  			return;
94  		}
95  
96  		if (heldFunds == null) {
97  			player.sendMessage(L.getString("transaction.error.economy"));
98  			return;
99  		}
100 
101 		if (!heldFunds.type.equals(ResponseType.SUCCESS)) {
102 			player.sendMessage(L.getString("transaction.error.economy"));
103 			return;
104 		}
105 
106 		// Check for availability
107 		Integer currentInvAmount = shop.getInventoryInterface().getInventoryStock(invItem.getItemStack());
108 
109 		if (baseItem.getAmount() > currentInvAmount) {
110 			if (charge) {
111 				UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
112 			}
113 			// Some error checks for deposit
114 			player.sendMessage(L.getString("transaction.sale.shop.notenoughitems"));
115 			return;
116 		} else {
117 			shop.getInventoryInterface().removeInventory(invItem.getItemStack(), buyStack.getAmount());
118 		}
119 
120 		//ItemStack placedStack = invItem.getItemStack().clone();
121 		//placedStack.setAmount(buyStack.getAmount());
122 		HashMap<Integer, ItemStack> returnedItems = player.getInventory().addItem(baseItem);
123 
124 		BigDecimal refund = BigDecimal.ZERO;
125 		Integer finalAmount = buyStack.getAmount();
126 
127 		if (!returnedItems.isEmpty()) {
128 			// calculate returned funds... readd to merchant
129 			Integer returned = 0;
130 			for (ItemStack returnedItem : returnedItems.values()) {
131 				returned = returned + returnedItem.getAmount();
132 			}
133 
134 			if (charge) {
135 				refund = buyStackPriceEach.multiply(BigDecimal.valueOf(returned));
136 
137 				UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), refund.doubleValue());
138 			}
139 
140 			ItemStack itemReturned = baseItem.clone();
141 			itemReturned.setAmount(returned);
142 			shop.getInventoryInterface().addInventory(itemReturned);
143 			finalAmount =- returned;
144 		}
145 
146 		BigDecimal traderDeposit = BigDecimal.valueOf(heldFunds.amount).subtract(refund);
147 
148 		// Deposit into trader when interface is ready.
149 		if (!charge) {
150 			player.sendMessage(L.getFormatString("transaction.sale.shop.purchase", buyStack.getType().name(), finalAmount));
151 			player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", "FREE"));
152 		} else if (shop.getWallet().addFunds(traderDeposit).type.equals(ResponseType.SUCCESS)) {
153 			player.sendMessage(L.getFormatString("transaction.sale.shop.purchase", buyStack.getType().name(), finalAmount));
154 			player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", UltraTrader.getEconomy().format(traderDeposit.doubleValue())));
155 		} else {
156 			player.sendMessage(L.getString("transaction.error.fundstoshop"));
157 		}
158 
159 		shop.save(true);
160 
161 		if (UltraTrader.isLoggingEnabled()) {
162 			UltraTrader.getLogDbObj().doLog(shop, player, new EconomyResponse(traderDeposit.doubleValue(), 0, ResponseType.SUCCESS, ""), ShopAction.SELL, "Player pruchased " + finalAmount + " " + buyStack.getType().name());
163 		}

173 	public static void processSale(Shop shop, final Player player, final ItemStack item) {
174 		ItemPrice invItem = shop.getItemPrice(item);
175 		BigDecimal buyStackPriceEach = invItem.getBuyPrice();
176 		BigDecimal buyStackPrice = buyStackPriceEach.multiply(BigDecimal.valueOf(item.getAmount()));
177         ItemStack removeStack = invItem.getItemStack().clone();
178         removeStack.setAmount(item.getAmount());
179 
180 		boolean success = true;
181 
182 		Wallet wallet = shop.getWallet();
183 
184         ShopPurchaseFromPlayerEvent event = new ShopPurchaseFromPlayerEvent(player, invItem, shop, removeStack, buyStackPrice);
185         Bukkit.getServer().getPluginManager().callEvent(event);
186 
187         if (event.isCancelled()) {
188             return;
189         }
190 
191 		EconomyResponse heldFunds = wallet.removeFunds(buyStackPrice);
192 
193 		if (!heldFunds.type.equals(ResponseType.SUCCESS)) {
194 			player.sendMessage(L.getString("transaction.sale.shop.notenoughfunds"));
195 			success = false;
196 		}
197 
198         if (player.getInventory().containsAtLeast(invItem.getItemStack(), item.getAmount())) {
199             player.getInventory().removeItem(removeStack);
200         } else {
201             success = false;
202         }
203 
204 		if (!success) {
205 			player.getInventory().addItem(removeStack);
206 			return;
207 		}
208 
209 		// Add the Item to Trader Inventory
210 		shop.getInventoryInterface().addInventory(removeStack);
211 
212 		EconomyResponse depResponse = UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
213 
214 		if (depResponse == null) {
215 			player.sendMessage(L.getString("transaction.error.economy") + 3);
216 			return;
217 		}
218 
219 		if (!depResponse.type.equals(ResponseType.SUCCESS)) {
220 			player.sendMessage(L.getString("transaction.error.economy") + 4);
221 			return;
222 		}
223 
224 		player.sendMessage(L.getFormatString("transaction.sale.shop.sale", item.getType().name(), item.getAmount()));
225 		player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", UltraTrader.getEconomy().format(heldFunds.amount)));
226 
227 		shop.save(true);
228 
229 		if (UltraTrader.isLoggingEnabled()) {
230 			UltraTrader.getLogDbObj().doLog(shop, player, heldFunds, ShopAction.BUY, "Sold " + item.getAmount() + " " + item.getType().name());
231 		}
232 	}
