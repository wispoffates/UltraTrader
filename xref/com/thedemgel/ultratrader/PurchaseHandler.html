13  /**
14   * If there is anything left (no order is required) a reverse order will be
15   * done, and price will be reduced by anything that remains.
16   *
17   * Order for secure transactions:
18   *
19   * 1) figure price for all items in "Buy Box" - "Buy Stack" 2) Check if player
20   * has available funds 3) Withdraw funds - hold funds in temp variable * 4)
21   * Remove items (also check for availability) from traders inventory * 5) If
22   * difference, remove from players "Buy Box" 6) Attempt to put all items in
23   * players inventory 7) Any remaining items not put into inventory - figure
24   * price value 8) subtract remaining value from holding 9) refund funds to
25   * player account 10) deposit funds into trader account 11) return any remaining
26   * items to trader inventory.
27   */

31  	 * Process a Sale TO A Player from a Shop.
32  	 *
33  	 * @param shop The shop the sale is coming from
34  	 * @param player The player the sale is going to
35  	 * @param buyStack The itemstack of the item clicked.
36  	 */
37  	public static void processPurchase(Shop shop, Player player, ItemStack buyStack) {
38  		String buyStackId = shop.getItemId(buyStack);
39  		ItemPrice invItem = shop.getSellprices().get(buyStackId);
40  		BigDecimal buyStackPriceEach = invItem.getPrice();
41  
42  		BigDecimal buyStackPrice = buyStackPriceEach.multiply(BigDecimal.valueOf(buyStack.getAmount()));
43  
44  		EconomyResponse heldFunds;
45  		if (UltraTrader.getEconomy().has(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue())) {
46  			heldFunds = UltraTrader.getEconomy().withdrawPlayer(player.getName(), player.getWorld().getName(), buyStackPrice.doubleValue());
47  		} else {
48  			player.sendMessage(L.getString("transaction.sale.player.notenoughfunds"));
49  			return;
50  		}
51  
52  		if (heldFunds == null) {
53  			player.sendMessage(L.getString("transaction.error.economy"));
54  			return;
55  		}
56  
57  		if (!heldFunds.type.equals(ResponseType.SUCCESS)) {
58  			player.sendMessage(L.getString("transaction.error.economy"));
59  			return;
60  		}
61  
62  		// Check for availability
63  		ItemStack baseItem = invItem.getItemStack();
64  		Integer currentInvAmount = shop.getInventoryInterface().getInventoryStock(baseItem);
65  
66  		if (buyStack.getAmount() > currentInvAmount) {
67  			UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
68  			// Some error checks for deposit
69  			player.sendMessage(L.getString("transaction.sale.shop.notenoughitems"));
70  			return;
71  		} else {
72  			shop.getInventoryInterface().removeInventory(baseItem, buyStack.getAmount());
73  		}
74  
75  		ItemStack placedStack = invItem.getItemStack().clone();
76  		placedStack.setAmount(buyStack.getAmount());
77  		HashMap<Integer, ItemStack> returnedItems = player.getInventory().addItem(placedStack);
78  
79  		BigDecimal refund = BigDecimal.ZERO;
80  		if (!returnedItems.isEmpty()) {
81  			// calculate returned funds... readd to merchant
82  			Integer returned = 0;
83  			for (ItemStack returnedItem : returnedItems.values()) {
84  				returned = returned + returnedItem.getAmount();
85  			}
86  
87  			refund = buyStackPriceEach.multiply(BigDecimal.valueOf(returned));
88  
89  			UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), refund.doubleValue());
90  
91  			ItemStack itemReturned = baseItem.clone();
92  			itemReturned.setAmount(returned);
93  			shop.getInventoryInterface().addInventory(itemReturned);
94  		}
95  
96  		BigDecimal traderDeposit = BigDecimal.valueOf(heldFunds.amount).subtract(refund);
97  
98  		// Deposit into trader when interface is ready.
99  		if (shop.getWallet().addFunds(traderDeposit).type.equals(ResponseType.SUCCESS)) {
100 			//player.sendMessage(MessageFormat.format(UltraTrader.getResourceBundle().getString("transaction.sale.shop.totalpurchase"), traderDeposit));
101 			player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", UltraTrader.getEconomy().format(traderDeposit.doubleValue())));
102 		} else {
103 			player.sendMessage(L.getString("transaction.error.fundstoshop"));
104 		}
105 
106 		shop.save();
107 	}

122 
123 		EconomyResponse heldFunds;

125 			heldFunds = wallet.removeFunds(buyStackPrice);
126 		} else {
127 			player.sendMessage(L.getString("transaction.sale.player.notenoughfunds"));
128 			return;
129 		}
130 
131 		if (heldFunds == null) {
132 			player.sendMessage(L.getString("transaction.error.economy") + 1);
133 			return;
134 		}
135

137 			player.sendMessage(L.getString("transaction.error.economy") + 2);
138 			return;
139 		}
140 
141 		// Add the Item to Trader Inventory
142 		shop.getInventoryInterface().addInventory(item);
143 
144 		EconomyResponse depResponse = UltraTrader.getEconomy().depositPlayer(player.getName(), player.getWorld().getName(), heldFunds.amount);
145 
146 		if (depResponse == null) {
147 			player.sendMessage(L.getString("transaction.error.economy") + 3);
148 			return;
149 		}
150 
151 		if (!depResponse.type.equals(ResponseType.SUCCESS)) {
152 			player.sendMessage(L.getString("transaction.error.economy") + 4);
153 			return;
154 		}
155 
156 		player.sendMessage(L.getFormatString("transaction.sale.shop.totalpurchase", UltraTrader.getEconomy().format(heldFunds.amount)));
