34  			p.sendRawMessage(prefix.getPrefix(context) + "Access: Enabled");
35  			p.sendRawMessage(prefix.getPrefix(context) + "Item Cost: " + UltraTrader.getEconomy().format(view.getShop().getRemoteItemCost()));
36  			addOption("disable", new AdminRemoteTogglePrompt(true));
37  		} else {
38  			p.sendRawMessage(prefix.getPrefix(context) + "Access: Disabled");
39  			p.sendRawMessage(prefix.getPrefix(context) + "Activation Cost: " + UltraTrader.getEconomy().format(LimitHandler.getRemoteActivateCost(p)));
40  			addOption("enable", new AdminRemoteTogglePrompt(false));
41  		}
42
